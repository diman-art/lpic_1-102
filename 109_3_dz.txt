Перед выполнением заданий важно самостоятельно повторить шаги, выполняемые на уроке.

1) Подключить несколько сетевых карт и указать верный маршрут в интернет по умолчанию.
# В случае замены сетевой карты просто оключи в этом же файле первую:
nano /etc/network/interfaces
# В строку auto допишем интерфейс новый, который будет автоматически запускаться
auto lo wlan1 ens0s5
metric 100

iface wlan1 inet dhcp	# wlan1 от dhcp получает адресс
iface ens0s5 inet static    # ens0s5 имеет статический адрес
address 192.168.0.5
natmask 255.255.255.0
gateway 192.168.0.10
metric 101

#Введи в терминал:
service networking restart
route

2) Получить информацию о DNS сайта it-semaev.ru.
host it-semaev.ru

3) Узнать ip-адрес серверов skype.
dig skype.com

4) Выяснить адреса маршрутизаторов, через которые проходят пакеты от вашего ПК до узла google.com.
traceroute google.com

5) Собрать статистику отправки сотни пакетов на сайт mail.ru.
ping  mail.ru -c 100 

6) Просмотреть все активные соединения при использовании любого мессенджера (скайп, icq и т.д.).
netstat -tulnp | grep icq

7) Отправить файл с одного ПК на другой при помощи nc.
apt-get install netcat
на стороне сервера
nc -l -p 3333 > file.txt

на стороне клиента:
cat file.txt | nc -q 0 <remote_ip> 3333
Ключ -q 0 нужен для того, что бы netcat завершил работу сразу после отправки

# можно отправлять файлы предварительно запаковав их:
на стороне сервера
nc -l -p 3333 | tar xv

на стороне клиента:
tar cv * | nc -q 0 <remote_ip> 3333
