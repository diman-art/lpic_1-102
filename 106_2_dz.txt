Перед выполнением заданий важно самостоятельно повторить шаги, выполняемые на уроке.

1) Установить все версии менеджера дисплея и включить каждый из них по очереди.
apt-get install gmd kdm xdm -y
# gdm выбираем по умолчанию

2) Настроить приветствтие в каждом DM.
sudo nano /etc/dconf/db/gdm.d/01-banner-message
# Впиши:
[org/gnome/login-screen]
banner-message-enable=true
banner-message-text='Привет'

# В терминал пиши:
sudo dconf update
